54 static inline uint64_t buf_get_bits(Buffer *buf, unsigned nb_bits) {
55  uint64_t tmp = 0;
56  assert(nb_bits <= 56);
57  if (nb_bits == 0)
58  return 0;
59  unsigned bytes = buf->index >> 3,
60  bits = buf->index & 7,
61  read_bytes = (nb_bits >> 3) + 2,
62  inv_read = 8 - read_bytes;
63  for (unsigned i = 7; i >= inv_read; --i) {
64  if (i == inv_read && (nb_bits & 7) <= 8 - bits)
65  break;
66  tmp |= (int64_t)buf->ptr[bytes++] << (i << 3);
67  }
68 
69  buf->index += nb_bits;
70 
71  return (tmp << bits) >> (64 - nb_bits);
72 }

125 static inline SWFError buf_append(Buffer *buffer, const uint8_t *add, size_t size) {
126  SWFError ret = SWF_OK;
127  if (!buffer->alloc_ptr)
128  if ((ret = buf_init(buffer, size)))
129  return ret;
130  uint8_t *end = buffer->alloc_ptr + buffer->alloc_size;
131  if (buffer->ptr + buffer->size + size <= end) {
132  buf_append_raw(buffer, add, size);
133  return SWF_OK;
134  }
135  if (buffer->alloc_ptr + buffer->size + size <= end) {
136  // We have a buffer, and our new data will fit after shuffling.
137  buf_shift(buffer);
138  buf_append_raw(buffer, add, size);
139  return SWF_OK;
140  }
141  // We have a buffer, but it's not big enough. Allocate a new one.
142  if ((ret = buf_grow_by(buffer, size)))
143  return ret;
144  buf_append_raw(buffer, add, size);
145  return SWF_OK;
146 }
