68 static SWFError parse_swf_header(SWFParser *parser){
69  if(parser->buf.size < 8){
70  return SWF_NEED_MORE_DATA;
71  }
72  SWF* swf = parser->swf;
73 
74  swf->compression = get_8(&parser->buf);
75  if(check_header(parser))
76  return set_error(parser, SWF_INVALID, "parse_swf_header: check_header reported an invalid header");
77  swf->version = get_8(&parser->buf);
78  swf->size = get_32(&parser->buf);
79  parser->state = PARSER_HEADER;
80  if(parser->callbacks.header_cb){
81  parser->callbacks.header_cb(parser, NULL, parser->callbacks.ctx);
82  }
83  return setup_decompression(parser);
84 }

86 static SWFError parse_swf_rect(SWFParser *parser, SWFRect* rect){
87  Buffer *buf = &parser->buf;
88  if(buf->size < 1)
89  return SWF_NEED_MORE_DATA;
90  int size = buf_get_bits(buf, 5);
91  if(buf->size < (size * 4 + 5 + 7) >> 3)
92  return SWF_NEED_MORE_DATA;
93  rect->x_min = buf_get_sbits(buf, size);
94  rect->x_max = buf_get_sbits(buf, size);
95  rect->y_min = buf_get_sbits(buf, size);
96  rect->y_max = buf_get_sbits(buf, size);
97  buf_finish_bit_access(buf);
98  return SWF_OK;
99 }

136  if(!tag->size)

168 static SWFError parse_tag(SWFParser *parser){
169  buf_clear_rollback(&parser->buf);
170  if(parser->buf.size < 2)
171  return SWF_NEED_MORE_DATA;
172  uint16_t code_and_length = get_16(&parser->buf);
173  uint32_t len = code_and_length & 0x3F;
174  uint16_t code = code_and_length >> 6;
175  if(parser->buf.size < len){
176  buf_rollback(&parser->buf);
177  return SWF_NEED_MORE_DATA;
178  }
179  if(len == 0x3F){
180  len = get_32(&parser->buf);
181  if(len > parser->buf.size){
182  buf_rollback(&parser->buf);
183  return SWF_NEED_MORE_DATA;
184  }
185  }
186  SWFTag tag = {
187  .type = code,
188  .payload = NULL,
189  .size = len,
190  .id = 0,
191  };
192  SWFError ret = SWF_OK;
193  switch(code){
194  case SWF_JPEG_TABLES:
195  if((ret = parse_JPEG_tables(parser, &tag))){
196  buf_rollback(&parser->buf);
197  return ret;
198  }
199  break;
200  case SWF_DEFINE_BITS:
201  case SWF_DEFINE_BITS_JPEG_2:
202  case SWF_DEFINE_BITS_JPEG_3:
203  case SWF_DEFINE_BITS_JPEG_4:
204  case SWF_DEFINE_BITS_LOSSLESS:
205  case SWF_DEFINE_BITS_LOSSLESS_2:
206  if((ret = parse_id_payload(parser, &tag))){
207  buf_rollback(&parser->buf);
208  return ret;
209  }
210  break;
211  case SWF_END:
212  parser->state = PARSER_FINISHED;
213  buf_advance(&parser->buf, len);
214  if(parser->callbacks.end_cb){
215  parser->callbacks.end_cb(parser, NULL, parser->callbacks.ctx);
216  }
217  return SWF_FINISHED;
218  default:
219  if((ret = parse_payload(parser, &tag))){
220  buf_rollback(&parser->buf);
221  return ret;
222  }
223  break;
224  }
225  if(parser->callbacks.tag_cb){
226  return parser->callbacks.tag_cb(parser, &tag, parser->callbacks.ctx);
227  }
228  return copy_error(parser, parser->swf, swf_add_tag(parser->swf, &tag));
229 }

244 static SWFError parse_buf(SWFParser *parser){
245  int was_ok = 0;
246  SWFError ret = SWF_OK;
247  while(ret == SWF_OK){
248  ret = parse_buf_inc(parser);
249  if(ret == SWF_OK)
250  was_ok = 1;
251  }
252  if(ret == SWF_NEED_MORE_DATA && was_ok)
253  return SWF_OK;

257 SWFError swf_parser_append(SWFParser *parser, const void *buf_in, size_t len){
258  SWF *swf = parser->swf;
259  const uint8_t *buf = buf_in;
260  SWFError ret = SWF_OK;
261  if(parser->state == PARSER_STARTED){
262  // This is a really stupid setup to make sure we don't crash
263  // if the first few packets are <8 bytes
264  uint8_t bytes_left = 8 - parser->buf.size;
265  bytes_left = bytes_left > 8 ? 8 : bytes_left;
266  bytes_left = len < bytes_left ? len : bytes_left;
267  if(!parser->buf.alloc_ptr &&
268  (ret = buf_init_with_data(&parser->buf, buf, bytes_left)))
269  return copy_error(parser, &parser->buf, ret);
270  if(parser->buf.size < 8)
271  return SWF_OK;
272  ret = parse_swf_header(parser);
273  if(ret != SWF_OK)
274  return ret;
275  // This buf is really tiny, and probably not worth keeping
276  buf_free(&parser->buf);
277  buf += bytes_left;
278  len -= bytes_left;
279  // Fall through to the post-header parsing stage
280  }
281  switch(swf->compression){
282  case SWF_UNCOMPRESSED:
283  if((ret = buf_append(&parser->buf, buf, len)))
284  return copy_error(parser, &parser->buf, ret);
285  return parse_buf(parser);
286  case SWF_ZLIB:
287  if(!parser->buf.alloc_ptr)
288  // No buffer yet. Allocate one, wild-guessing at the size
289  if((ret = buf_init(&parser->buf, len * 4)))
290  return copy_error(parser, &parser->buf, ret);
291  int increase_space = 0;
292  parser->zstrm.avail_in = len;
293  parser->zstrm.next_in = (uint8_t*)buf;
294  for(;;){
295  size_t avail_size = buf_shift(&parser->buf);
296  if(!avail_size && (ret = buf_grow_by(&parser->buf, parser->zstrm.avail_in * 4)))
297  return copy_error(parser, &parser->buf, ret);
298  parser->zstrm.avail_out = avail_size;
299  parser->zstrm.next_out = parser->buf.ptr + parser->buf.size;
300  int z_ret = inflate(&parser->zstrm, Z_NO_FLUSH);
301  parser->buf.size += (avail_size - parser->zstrm.avail_out);
302  switch(z_ret){
303  case Z_STREAM_END:
304  return parse_buf(parser);
305  case Z_DATA_ERROR:
306  return set_error(parser, SWF_INVALID, "swf_parser_append: inflate() returned Z_DATA_ERROR");
307  case Z_STREAM_ERROR:
308  return set_error(parser, SWF_INTERNAL_ERROR, "swf_parser_append: inflate() returned Z_STREAM_ERROR");
309  case Z_MEM_ERROR:
310  return set_error(parser, SWF_NOMEM, "swf_parser_append: inflate() returned Z_MEM_ERROR");
311  case Z_BUF_ERROR:
312  increase_space = 1;
313  case Z_OK:
314  // Continue looping
315  break;
316  default:
317  return set_error(parser, SWF_UNKNOWN, "swf_parser_append: inflate() returned an unknown error");
318  }
319  switch((ret = parse_buf(parser))){
320  case SWF_OK:
321  break;
322  case SWF_NEED_MORE_DATA:
323  if(increase_space){
324  if((ret = buf_grow(&parser->buf, 2))){
325  return copy_error(parser, &parser->buf, ret);
326  }
327  increase_space = 0;
328  continue;
329  }
330  default:
331  return ret;
332  }
333  if(!parser->zstrm.avail_in){
334  return ret;
335  }
336  }
337  default:
338  return set_error(parser, SWF_UNKNOWN, "swf_parser_append: inflate() returned an unknown error");
339  }
340 }
